function addLeadFromTracker(name, email){
  if(!email) return;
  const exists = getAllLeads().find(l=>l.email===email||(l.notes&&l.notes.includes(email)));
  if(exists){ alert('This lead is already in your tracker!'); return; }
  S.leads.push({
    id: Date.now(),
    name: name||email,
    platform: 'Quiz Funnel',
    service: 'Not Sure Yet',
    status: 'New',
    followup: new Date(Date.now()+86400000*2).toISOString().split('T')[0],
    budget: '',
    notes: 'Quiz completion · email: '+email,
  });
  save(); renderAll();
  alert('Added to Lead Tracker! Check your Leads tab.');
}

function clearTrafficData(){
  if(!confirm('Clear all traffic data? This cannot be undone.')) return;
  localStorage.removeItem('de-tracker');
  localStorage.removeItem('de-wp-tracker');
  renderTraffic();
}

function showSnippet(type){
  const el = document.getElementById('snippet-display');
  const title = document.getElementById('snippet-title');
  const code  = document.getElementById('snippet-code');
  const instr = document.getElementById('snippet-instructions');
  if(type==='quiz'){
    title.innerHTML = '📋 Quiz Tracking Snippet';
    code.textContent = QUIZ_SNIPPET;
    instr.innerHTML  = '📍 Where to paste: Open your quiz HTML file in Netlify. Find the closing </body> tag. Paste this snippet just before it. Redeploy your site.';
  } else if(type==='wp'){
    title.innerHTML = '📋 WordPress Tracking Snippet';
    code.textContent = WP_SNIPPET;
    instr.innerHTML  = '📍 Where to paste: In WordPress admin → Install "Code Snippets" plugin → Add New → paste this code → set to run everywhere → Save & Activate.';
  } else {
    title.innerHTML = '📋 Netlify Function';
    code.textContent = NETLIFY_SNIPPET;
    instr.innerHTML  = '📍 Where to add: In your quiz repo, create folder netlify/functions/ and save this as track.js. Redeploy your Netlify site.';
  }
  el.style.display = 'block';
  el.scrollIntoView({behavior:'smooth'});
}

function copySnippet(){
  const code = document.getElementById('snippet-code').textContent;
  navigator.clipboard.writeText(code).then(()=>alert('Copied! Paste it into your file.')).catch(()=>{
    const ta = document.createElement('textarea');
    ta.value = code; document.body.appendChild(ta); ta.select();
    document.execCommand('copy'); document.body.removeChild(ta);
    alert('Copied!');
  });
}
